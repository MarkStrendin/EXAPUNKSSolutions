LINK 800
;USE FILE TO TRACK ROW
MAKE
COPY 0 F

;F = ROW
;X = COL
;M = DATA
;T = COMPARE


MARK LSTART
 SEEK -1
 TEST F = 3
 TJMP END

 SEEK -1
 ;ADD THE CHARACTER
 COPY F #DATA 
 COPY X #DATA
 COPY M #DATA 

 ;INC COUNTER
 ADDI X 1 X
 
 ;IF COUNTER = MAX CHARS
 ;FOR THE ROW, INC ROW
 ;AND RESET COUNTER
 TEST X > 8 
 FJMP LSTART

 COPY 0 X
 SEEK -1
 ADDI 1 F F
JUMP LSTART 
MARK END
WIPE
